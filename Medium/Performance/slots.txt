01. Use __slots__ to reduce memory overhead in classes
BEFORE
class Animal:
    def __init__(self, id, species, locations):
        self.id = id
        self.species = species
        self.locations = locations  # List of GPS tuples

AFTER
class Animal:
    __slots__ = =['id', 'species', 'locations']
    def __init__(self, id, species, locations):
        self.id = id
        self.species = species
        self.locations = locations  # List of GPS tuples
